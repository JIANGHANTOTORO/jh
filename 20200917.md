//练习4-11 统计素数并求和

#include<stdio.h>
#include<math.h>
int isPrime(int n){
    int flag = 1, i;
    if(n == 1){
        flag = 0;
    }
    for(i = 2; i <=sqrt(n); i++){
        if(n % i == 0){
            flag = 0;
        }
    }
    return flag;
}
int main(void){
    int M, N, count = 0, sum = 0;
    scanf("%d %d", &M, &N);
    for(int i = M; i <= N; i++){
        if(isPrime(i)){
            count ++;
            sum = sum + i;
        }
    }
    printf("%d %d",count,sum);
    return 0;
}
