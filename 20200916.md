//练习2-18 求组合数

#include<stdio.h>
double fact(int n){
    double product = 1;
    int i;
    for(i = 1; i <= n; i++){
        product = product * i;
    }
    return product;
}
int main(void){
    int m, n;
    double result = 1;
    scanf("%d %d",&m, &n);
    for(int i = 0; i <= n; i++){
        result = fact(n)/(fact(m)*fact(n-m));
    }
   printf("result = %0.f",result);
   return 0;
}
