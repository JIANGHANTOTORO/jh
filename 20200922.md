//习题7-2 求一批整数中出现最多的个位数字

#include <stdio.h>
int main() {  

    int n,i;  
    int number[10]={0};
	int max = 0;  
    int a;  
    scanf("%d", &n);  
    for (i=0; i<n; i++) {  
        scanf("%d", &a);  
        while (a > 0) {  
            number[a%10]++;  //将数的每一位存进数组
            a /= 10;  
        }  
    }  
    //找出数组中最大的数
    for (i=0; i<10; i++) {  
        if (max < number[i]) {  
            max = number[i];  
        }  
    }  
      
    printf("%d:", max);  
    for (i=0; i<10; i++) {  
        if (max == number[i]) {  
            printf(" %d",i);  
        }  
    }  
      
    return 0;  

}

//习题7-3 判断上三角矩阵

#include <stdio.h>  
int main() {  
	int t,n,i,j,k;
	int a[10][10]; 
	scanf("%d",&t);
	int count=0;
	int b[100]={0};  //记录判断记录 
	
	for(k=0; k<t; k++){
		//读入矩阵 
		scanf("%d",&n);
		for(i=0; i<n; i++){
			for(j=0; j<n; j++){
				scanf("%d",&a[i][j]);	
			}		
		}
		//判断矩阵 
		for(i=0; i<n; i++){
			for(j=0; j<n; j++){
				if(i>j && a[i][j] != 0){ //判断主对角线
					count++;
				}
			}		
		}
		if(count==0 ){
    		b[k] = 1;
    	}else{
    		b[k] = 0;
    	}
    	count=0;		
	}
	
	for(k=0; k<t; k++){
		if(b[k]==1){
			printf("YES\n");
		}else{
			printf("NO\n");
		} 
	}
	
    return 0;  
}
