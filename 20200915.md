//习题2-3 求平方与倒数序列的部分和
#include<stdio.h>
int main(void)
{
int i, n, m;
	double sum;
	scanf("%d%d", &m, &n);
	sum=0;
	for(i=m; i<=n; i++){
		sum=sum+i*i+1.0/i;
	}
	printf("sum = %.6f", sum);
	
	return 0;
}

//习题2-4 求交错序列前N项和
//这里存在测试点，测试输入的是奇数还是偶数，以及最小奇数和最小偶数
#include<stdio.h>
int main(void){
int i, flag, b, d, n;
	double sum;
	scanf("%d", &n);
	sum=0; 
	flag=1;
	d=1;
	b=1;
	for(i=1; i<=n; i++){
		sum=sum+b*flag*1.0/d;
		flag=flag+1;
		b=-b;
		d=d+2;
	}
	printf("%.3f", sum);
	
	return 0;
    
}
