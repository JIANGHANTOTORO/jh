//习题6-7 简单计算器

#include<stdio.h>
int main(void){
    int i, sum, is = 0;
    char op = '0';
    scanf("%d",&sum);
    while(op != '='){
        scanf("%c",&op);
        if(op == '='){
            break;
        }
        scanf("%d",&i);
        if(op == '+'){
            sum += i;
        }else if(op == '-'){
            sum -= i;
        }else if(op == '*'){
            sum *= i;
        }else if(op == '/'){
            if(i != 0){
                sum /= i;
            }else{
                is=1;
            }
        }else{
            is=1;
        }
    }
    if(is==1){
        printf("ERROR\n");
    }else{
        printf("%d\n",sum);
        return 0;
    }
}

//习题6-8 统计一行文本的单词个数

#include<stdio.h>
#include<string.h>
int main()
{
  char a[1000];
  gets(a);//输入字符串 
  int m,num;
  m=strlen(a);//字符串长度 
  if(a[0]==' ')
      num=0;
  else
      num=1;
  for(int i=0;i<m-1;i++)//最后一个字符为结束符 
  {
      if(a[i]==' ' && a[i+1]!=' ')
          num++;
  }
  printf("%d\n",num);
  return 0; 
}
